		function changeLinks () {
			if (vp.activityData['openhouse'] === 'true') {
				$('a').each(function () {
					$a = $(this);
					var newLink = $a.attr('data-href-o');

				});

					$a = $(this);

					if ( originalLink ) {
						$a.attr('href', originalLink);
					}
				});
				clearInterval(ohInterval);
			}
