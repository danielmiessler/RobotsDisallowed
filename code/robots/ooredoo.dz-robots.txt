    function checkSubMenu(sm) {
        var allimagesloaded = false;
        var imgs = sm.find('img');
        if (imgs.length == 0) {
            allimagesloaded = true;
        } else {
            allimagesloaded = true;
            for (var i=0 ; i<imgs.length; i++) {
                if (imgs.eq(i).width() == 0) allimagesloaded = false;
            }
        }
        // Solo comprueba cuando las imágenes están guardadas
        if (allimagesloaded) {
            var pt = sm.offset().left + sm.outerWidth();
            var ww = $(window).width();
            console.log(pt + ',' + $(window).width());
            if (pt > ww) {
                sm.css('marginLeft', $(window).width() - pt);
            }
        } else {
            setTimeout(function() {
                checkSubMenu(sm);
            }, 100);
        }
    }
