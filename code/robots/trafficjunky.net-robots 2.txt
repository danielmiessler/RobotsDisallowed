function switchDimensions( device ) {
    $(".page-zoneFormats .native-select-size, .animation-rules").hide();
    $(".page-zoneFormats .native-select-size[data-selected-device=" + device +"]").show();
    $(".page-zoneFormats .animation-rules").show();
    switchRules( device, $(".page-zoneFormats .native-select-size[data-selected-device=" + device +"]").val() );
}
function switchRules( device, dimension ) {
    $(".page-zoneFormats ul.animation-rules").empty();
    $.each(zoneFormatRules[device], function(index, value) {
        if ( index == dimension ) {
            var currentDescription = this;
            $.each( currentDescription, function( index,value ) {
                $.each( value,function( index,value1 ) {
                    $(".page-zoneFormats ul.animation-rules").append( '<li>' + value1 + '</li>');
                });
            });
        }
    });
}

function customFlashMessage( messageObject ) {
    // Content and appearance
    var type = messageObject.type || 'warning'; // Default yellow message if not specified; type: 'success'|'danger'|'warning'
    var message = messageObject.message;
    var stayOpen = messageObject.stayOpen ? 'stayOpen' : '';

    // Timing
    var showAtLeast = 5000; // Show message for at least
    var hideAfter = 12000; // Don't show a message for more than
    var hideDelay = Math.max(Math.min(message.length * 75, hideAfter), showAtLeast) || showAtLeast; // ms per character in message, with fallback

    var hasLink = message.indexOf('<a') > -1;

    // Messages containing links should stay open (unless explicitly told not to stay open)
    if(hasLink && messageObject.stayOpen !== false) stayOpen = 'stayOpen';

    // Create message DOM element
    var $flashMessage = $("<div>", {
        'class'           : "flash-message" + " " + stayOpen,
        'data-hide-delay' : hideDelay,
    });
    if (Boolean(messageObject.attributes)) {
        $flashMessage.attr( messageObject.attributes );
    }

    $flashMessage.addClass( type );
    $flashMessage.append("\
        <div class='flash-message-type'></div>\
        <div class='flash-message-text'></div>\
        <div class='message-close'><div class='svg-message-close'></div></div>");
    $flashMessage.find(".flash-message-text").html( message );
    switch ( type ) {
        case 'success':
            $(".flash-message-type", $flashMessage).addClass("fa fa-check");
            $(".messagesContainerTour .success-messages").append($flashMessage);
            break;
        case 'danger':
            $(".flash-message-type", $flashMessage).addClass("fa fa-exclamation");
            $(".messagesContainerTour .danger-messages").append($flashMessage);
            break;
        case 'warning':
            $(".flash-message-type", $flashMessage).addClass("fa fa-info");
            $(".messagesContainerTour .warning-messages").append($flashMessage);
            break;
    }


    // Close message after x mseconds
    flashMessageTimeout($flashMessage);

    var clickPosition;
    var getClickPosition = function(e) {
        return e.pageX + ',' + e.pageY;
    };

    // Hide on click
    $flashMessage.mousedown(function(e) {
        clickPosition = getClickPosition(e);
    });

    $flashMessage.mouseup(function(e) {
        // If the mouse moved do not dismiss the message because the user may be selecting text in the banner.
        // This effectively ignores 'drag-clicks'
        var clickReleasePosition = getClickPosition(e);
        if(clickReleasePosition !== clickPosition || e.which !== 1) return false;

        $(this).delay(100).slideUp(function() {
            $(this).remove();
        });
    });

    return $flashMessage;
}

function flashMessageTimeout(el) {
    var isStayOpen = $(el).is('.stayOpen');

    }

}

//Quick sign up, sign up form regex validation - username
.on("change keyup paste blur", ".page-signUp #username, .quickSignup  #username, .page-sem #username", function () {
    var $fieldErrorElement = $(this).parents("form").find(".field_error[data-error-field=username]");
    var value = $(this).val();
    if( !Boolean(value.length)) {
        $(this).addClass("invalid");
        $fieldErrorElement.html("<ul><li>Username can't be blank</li></ul>");
        if(Boolean($(".page-signUp .field_error ul").length)) {
            $(".page-signUp .signUpButton").attr("disabled", true);
        }

    } else if ( value.length < 3 || value.length > 15) {
        $(this).addClass("invalid");
        $fieldErrorElement.html("<ul><li>Please use between 3 and 15 characters</li></ul>");
        if(Boolean($(".page-signUp .field_error ul").length)) {
            $(".page-signUp .signUpButton").attr("disabled", true);
        }

    } else {
        var pattern = /^[a-zA-Z0-9_\\.]+(?:[_\\.-]?[a-zA-Z0-9_\\.]+)*$/,
        valid = Boolean( value.match( pattern ));
        if( !valid ) {
            $(this).addClass("invalid");
            $fieldErrorElement.html("<ul><li>Please use only letters (a-Z), numbers, underscores, and periods</li></ul>");
        } else {
            var firstLetter = value[0],
                lastLetter  = value[value.length - 1],
                matchArray  = ["_", "."];

            if ( matchArray.indexOf(firstLetter) >= 0 || matchArray.indexOf(lastLetter) >= 0) {
                $(this).addClass("invalid");
                $fieldErrorElement.html("<ul><li>Underscores and periods are only allowed to be used in the middle of your username</li></ul>");
            } else {
                $(this).removeClass("invalid");
                $fieldErrorElement.empty();
            }
        }
    }
})

//Quick sign up, sign up form regex validation - email
.on("change keyup paste blur",
    ".page-signUp #email, .quickSignup #email, .page-forgotAccount #email, .page-forgotAccount #activationEmail, .page-sem #email",
    function () {
        var elementId = $(this).attr("id");
        var $fieldErrorElement = $(this).parents("form").find(".field_error[data-error-field=" + elementId + "]");
        var value = $(this).val();
        if( !Boolean(value.length)) {
            $(this).addClass("invalid");
            $fieldErrorElement.html("<ul><li>Email can't be blank</li></ul>");

        } else {
            var pattern = /^([\w$&+,:;=?@#|'<>.^*()%!{}|~_-]*[@][a-zA-Z0-9][\w\-.]*[.][a-zA-Z0-9][\w.]+)$/,
                valid = Boolean( value.match( pattern ));
            if( !valid ) {
                $(this).addClass("invalid");
                $fieldErrorElement.html("<ul><li>Please enter a valid e-mail address</li></ul>");
            } else {
                $fieldErrorElement.empty();
                $(this).removeClass("invalid");
            }
        }
})

.on("blur keyup paste change",
    ".page-signUp #password, .passwordForm #password, .page-signUp #password_confirmation, .passwordForm #password_confirmation, .page-sem #password, .page-sem #password_confirmation",
    function() {
        var elementId = $(this).attr("id");
        var $fieldErrorElement = $(this).parents("form").find(".field_error[data-error-field=" + elementId + "]");
        $fieldErrorElement.empty();
        $(this).removeClass("invalid");
        var value = $(this).val();
        if( !Boolean(value.length)) {
            $(this).addClass("invalid");
            $fieldErrorElement.html("<ul><li>Password can't be blank</li></ul>");

        } else if ( value.length < 8) {
            $(this).addClass("invalid");
            $fieldErrorElement.html("<ul><li>The password must be at least 8 characters</li></ul>");

        } else if( elementId == 'password_confirmation') {
            var passwordInput = $(this).parents("form").find("#password").val();
            var match = (value == passwordInput);
            $(this).toggleClass("invalid", !match);
            if(match) {
                $fieldErrorElement.empty();
            } else  {
                $fieldErrorElement.html("<ul><li>Passwords do not match</li></ul>");
            }
        }
});
