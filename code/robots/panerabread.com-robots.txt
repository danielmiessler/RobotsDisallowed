        var filter = /^([\w-\.\+]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
        if (filter.test(email)) {
            return true;
        } else {
            return false;
        }
    }
    function vp(password) {
        "use strict";
        var filter = /^[a-zA-Z0-9_\-!@#$%^&*()+={}[\]\;\:\.\>\<\,\\\/\`\~\|]{6,20}$/;
        if (filter.test(password)) {
            return true;
        } else {
            return false;
        }
    }
