        function GetTLD() {
            if (!Bind('MainContent_hdnDomains', this, 'Text1', 1, 'MainContent_lblDomainSearchError')) {
                return false;

            }
            var i = 0;
            var label = "";
            var top = "";
            var _flag = false;
            $('#chkPopular input:checkbox').each(function () {
                if ($(this).is(':checked') == true) {
                    if (i == 0)
                        top = $('label[for=' + this.id + ']').html();
                    else
                        label += $('label[for=' + this.id + ']').html() + "|";

                    i++;
                }

            });
            $('#chkmore input:checkbox').each(function () {
                if ($(this).is(':checked') == true) {
                    if (i == 0 && top == "")
                        top = $('label[for=' + this.id + ']').html();
                    else
                        label += $('label[for=' + this.id + ']').html() + "|";

                    i++;
                }

            });
            label = label.substring(0, label.length - 1);

            if (top != "") {
                $("#MainContent_hdndomain").val($("#Text1").val());
                $("#MainContent_slTLD").val(top);
                _flag = true;
                if (label != "")
                    $("#MainContent_tld").val(label);

            }
            else {
                $("#MainContent_lblDomainSearchError").text("Please select tld from the list.");
                _flag = false;
            }
            return _flag;
        }
