function adjust(img){
  var imgContainer = img.parentNode,
      width = imgContainer.offsetWidth,
      height = imgContainer.offsetHeight,
      scale = (width / height).toFixed(2),
      ratio = (img.width / img.height).toFixed(2);
  if(Math.abs(scale - ratio) < 0.02){
      img.width = width;
      img.height = height;
  }else if(scale > ratio){
      img.width = width;
      img.height = img.width / ratio;
         // 图片上移
      img.style.marginTop = -(img.height - height) + "px" ;
  }else{
      img.height = height;
      img.width = img.height * ratio;
         // 图片右移
      img.style.marginLeft = -(img.width - width)/2 + "px";
  }
}
