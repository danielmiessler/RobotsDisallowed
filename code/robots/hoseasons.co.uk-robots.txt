//changing call center times
$(document).ready(function(){
    $("#callcentertimes").ready(function(){
        var rightNow = new Date();
        var day = rightNow.getDay();
        var hour = rightNow.getHours();
        var minutes = rightNow.getMinutes();
        var exactHour = hour + (minutes/60);
         
        var tempbank = new Date();
        tempbank.setHours(0);
        tempbank.setMinutes(0);
        tempbank.setSeconds(0);
        tempbank.setMilliseconds(0);
         
        //enter the bank holiday dates here.
        var newyearsday = new Date(2015, 0, 1),
        goodfriday = new Date(2015, 3, 3),
        eastermonday = new Date(2015, 3, 6),
        mayday = new Date(2015, 4, 4),
        springbankholiday = new Date(2015, 4, 25),
        summerbankholiday = new Date(2015, 7, 31),
        christmaseve = new Date(2015, 11, 24),
        christmasday = new Date(2015, 11, 25),
        boxingday = new Date(2015, 11, 26),
        newyearseve = new Date(2015, 11, 31);
         
        var callmessage = "until 9:30pm today:";
         
        // General Bank holidays
        if (tempbank.toDateString() == goodfriday.toDateString() || tempbank.toDateString() == eastermonday.toDateString() || tempbank.toDateString() == mayday.toDateString() || tempbank.toDateString() == springbankholiday.toDateString() || tempbank.toDateString() == summerbankholiday.toDateString()){
            if (exactHour >=9 && hour < 17){
                callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 5pm today"; 
            } else if(exactHour < 9){ /* morningclosed */
                callmessage = "Low call rates apply<br/> Open from 9:30am today"; 
            } else if(exactHour >= 17){ /* #eveningclosed  */ 
                callmessage = "Low call rates apply<br/> Open from 9:30am tomorrow"; 
            }
        } else if (tempbank.toDateString() == christmaseve.toDateString()) { /* Christmas Eve */
            if(exactHour >=9 && exactHour < 12){ /* open */ 
                callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 12pm"; 
            } else if (exactHour < 9){ /* morningclosed */ 
                callmessage = "Low call rates apply<br/> Open from 9am today"; 
            } else if (exactHour >= 12){ /* eveningclosed  */ 
                callmessage = "Low call rates apply<br/> Closed until 27th December"; 
            }
        } else if (tempbank.toDateString() == christmasday.toDateString() || tempbank.toDateString() == boxingday.toDateString()){ /* Christmas Day & Boxing Day */
            callmessage = "Low call rates apply<br/> Closed until 9am on 27th December";
        } else if (tempbank.toDateString() == newyearseve.toDateString()) { /* New Year's Eve */
            if (exactHour >=9 && exactHour < 17) { /* open */
                callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 5pm";
            } else if (hour < 9){ /* morningclosed */
                callmessage = "Low call rates apply<br/> Open from 9am today";
            } else if ( hour >= 17 ){ /* eveningclosed  */
                callmessage = "Low call rates apply<br/> Open from 10am tomorrow";
            }
        } else if (tempbank.toDateString() == newyearsday.toDateString()){ /* New Year's Day */
            if (hour >=10 && hour < 14){ /* open */
                callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 2pm"; 
            } else if (hour < 10){ /* morningclosed */
                callmessage = "Low call rates apply<br/> Open from 10am today"; 
            } else if(hour >= 14){ /* eveningclosed  */ 
                callmessage = "Low call rates apply<br/> Open from 9am tomorrow"; 
            }
        } else { /* Regular opening times */
            if (day >= 1 && day <= 4){
                if (hour >= 8 && exactHour < 22){ /* monday to thursday */
                    callmessage = "Low call rates apply, <b>call us now</b><br/>We're open until 10pm today"; 
                } else if (hour < 8){ /* morningclosed */
                    callmessage = "Low call rates apply<br/> Open from 8am today"; 
                } else if (exactHour >= 22){ /* eveningclosed */
                    callmessage = "Low call rates apply<br/> Open from 8am tomorrow"; 
                }
            }
            if (day == 5){
                if ( hour >= 8 && exactHour < 22 ) { /* friday */     
                    callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 10pm today"; 
                } else if (hour < 8){ /* morningclosed */
                    callmessage = "Low call rates apply<br/> Open from 8am today"; 
                } else if ( exactHour >= 22 ){ /* eveningclosed */
                    callmessage = "Low call rates apply<br/> Open from 9am tomorrow"; 
                }
            }
            if (day == 6){
                if (hour >= 9 && hour < 20){ /* saturday */
                    callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 8pm today";
                }
                if (hour < 9){ /* saturdaymorningclosed */
                    callmessage = "Low call rates apply<br/> Open from 9am today";
                }
                if (hour >= 20){ /* saturdayeveningclosed */
                    callmessage = "Low call rates apply<br/> Open from 9:30am tomorrow"; 
                }
            }
            if (day == 0){
                if (exactHour >= 9.5 && exactHour < 21.5){ /* sunday */
                    callmessage = "Low call rates apply - <b>call us now</b><br/>We're open until 9:30pm today";
                }
                if (exactHour < 9.5){ /* sundaymorningclosed */ 
                    callmessage = "Low call rates apply<br/> Open from 9:30am today"; 
                }
                if (exactHour >= 21.5){ /* eveningclosed  */
                    callmessage = "Low call rates apply<br/> Open from 8am tomorrow";
                }
            }
        }
         
        $("#callcentertimes").html(callmessage);
         
        if (day == 6 || day == 0){
            $("#opentoday").hide(); 
        } else {
            $("#opentoday").show(); 
        }
    });
});

$('[placeholder]').focus(function() {
  var input = $(this);
  if (input.val() == input.attr('placeholder')) {
    input.val('');
    input.removeClass('placeholder');
  }
}).blur(function() {
  var input = $(this);
  if (input.val() == '' || input.val() == input.attr('placeholder')) {
    input.addClass('placeholder');
    input.val(input.attr('placeholder'));
  }
}).blur();
 
$('[placeholder]').parents('form').submit(function() {
  $(this).find('[placeholder]').each(function() {
    var input = $(this);
    if (input.val() == input.attr('placeholder')) {
      input.val('');
    }
  })
});
