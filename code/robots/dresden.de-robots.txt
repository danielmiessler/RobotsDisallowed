                                        if($(this).attr("class") !== undefined &&amp; neighbor.attr("class") !== undefined)
                                        {
                                            if (($(this).attr("class").indexOf("teaser_image") === -1) &&amp; (neighbor.attr("class").indexOf("teaser_image") === -1))
                                            {
                                                //Höhen-Attribute auf 0 setzen, um Neurendern anzustoßen und anschließend entfernen
                                                $(this).height("auto");
                                                neighbor.height("auto");
                                                $(this).removeAttr("height");
                                                neighbor.removeAttr("height");

                                                if ($(document).width() >= 520 &&amp; $(document).width() < 900)
                                                {
                                                    //Höhe der benachbarten Elemente gleich setzen
                                                    if ($(this).height() > neighbor.height())
                                                        neighbor.height($(this).height());
                                                    else if ($(this).height() < neighbor.height())
                                                        $(this).height(neighbor.height());
                                                }
                                            }
                                        }
                                    }
                                }
                            });
                        });
                    });
                }
                //
            </script>

            function alignSkyscraper(){
                var winWidth = $(window).width();
                var pageObj=$("main.page");
                var adzone=$(".element_adzone");
                var posPage = pageObj.offset();
                var width = parseInt(pageObj.css("width"), 10);
                var offset = 9;
                var top = posPage.top;
                var left = posPage.left + width + offset;

                //bei Desktopansicht (1024px + 2 * 120px Skyscraper + 2* offset) neben dem Content anzeigen, sonst darunter
                if(winWidth >= (1024 + 2 * 120 + 2 * offset)){
                    adzone.css("position", "absolute");
                    adzone.offset({top: top, left: left});
                    adzone.removeClass("mobile");
                }
                else{
                    adzone.css("position", "relative");
                    adzone.css("top", offset);
                    adzone.css("left", "0");
                    adzone.css("margin", "0 auto");
                    adzone.addClass("mobile");
                }
            }

        </script>
