            var minTime= typeof(min)=="undefined"?"": new myTime(min);

            if(maxTime!="" && time.getSeconds()>maxTime.getSeconds()){
                $(this).val(maxTime.getString());
            }
            if(minTime!="" && time.getSeconds()<minTime.getSeconds()){

                $(this).val(minTime.getString());
            }
            normilizeTime();
            if($(this).attr("id")=="from_time"){
                ytPlayer.seekToStart();
            }else if($(this).attr("id")=="to_time"){
                ytPlayer.seekToEnd();
            }

            return;
        });
    }

    function normilizeTime(){

        var from_time=new myTime($("#from_time").val());
        var to_time=new myTime($("#to_time").val());
        //var max_time=new myTime($("#to_time").attr("max"));

        var from_seconds=from_time.getSeconds();
        var to_seconds=to_time.getSeconds();
        //var max_seconds=max_time.getSeconds;

        if(from_seconds>=to_seconds){
            from_seconds=to_seconds-1;
            if(from_seconds<0){
                from_seconds=0;
                to_seconds=1;
            }
            from_time=new myTime(from_seconds);
            to_time=new myTime(to_seconds);
            $("#from_time").val(from_time.getString());
            $("#to_time").val(to_time.getString());
        }

        ytPlayer.setPlayFrom(from_time.getSeconds());
        ytPlayer.setPlayTo(to_time.getSeconds());

        slider.noUiSlider.set([from_time.getSeconds(),to_time.getSeconds()]);

    }


    function secondsToTime(seconds){
        var minutes=Math.floor(seconds/60);
        seconds=seconds%60;

        seconds=Math.floor(seconds);

        if(seconds<10) seconds="0"+seconds;
        return minutes+":"+seconds;
    }

            if(slider.to_time!=to_time_sec){
                var to_time=new myTime(to_time_sec);
                $("#to_time").val(to_time.getString()).change();
            }
            slider.from_time=from_time_sec;
            slider.to_time=to_time_sec;

        });
    }
