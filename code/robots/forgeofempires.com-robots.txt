	/*
	 Processes the errors from PHP to JSON
	 Then checks for errors in the register form and dynamically creates error tooltip for each
	 formfield with an error.
	 */
	function displayRegisterFormErrors(json_errors) {
		var error_max_height = 34;

		$.each(json_errors, function (key, value) {
			key = String(key);

			var target_element;
			var error_message_element = $('<p class="error_message">' + value + '</p>');

			switch (key) {
				case 'email_already_in_use':
				case 'email_already_in_use_or_invalid':
				case 'email_invalid_characters':
					target_element = $('#form_element_email');
					break;
				case 'name_too_short_to_too_long':
				case 'name_invalid_characters':
				case 'name_forbidden':
				case 'name_already_in_use':
				case 'name_contains_adjacent_spaces':
					target_element = $('#form_element_name');
					break;
				case 'passwords_not_equal':
				case 'password_too_short':
				case 'password_not_safe':
				case 'password_contains_trailing_spaces':
				case 'password_equals_to_name':
					target_element = $('#form_element_password');
					break;
				case 'terms_not_accepted':
					target_element = $('#form_element_terms');
					break;
			}
			// When error_message_element already exists, just update the text value inside the dom element
			if (target_element) {
				if (!target_element.children('.error_message').length > 0) {
					target_element.append(error_message_element);
				} else {
					$('.error_message', target_element).text(value);
				}
			}
			// Ensure correct positioning and size, depending on oversized messages
			if (error_message_element.height() > error_max_height) {
				error_message_element.css({"font-size": "12px", "line-height": "12px", "padding": "3px 5px"});
			}

			// Inserts a graphical pointer-element (css3-based) into every error_message tooltip element
			var pointer_element = $('<div class="pointer" />');
			if ($('.error_message div', target_element).hasClass('pointer')) {
				return;
			}
			$('.error_message', target_element).prepend(pointer_element);
		});
	}
</script>
