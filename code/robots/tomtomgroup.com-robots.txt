   function onReady(method) {
      if (window.jQuery) {
         method();
      } else {
         setTimeout(function() { onReady(method) }, 50);
      }
   }

   // Desktop only
   onReady(function() {
      var isDesktop = window.innerWidth >= 1024;
      if (!isDesktop) {
           return;
      }

      var $nav = $('.tt-main-navigation-module');
      var $cat = $('.tt-homepage__cat');
      var $cards = $('.tt-homepage__cards');
      var $panels = $('.tt-panels');
      var $logoImg = $nav.find('.tt-logo img');

      $logoImg.get(0).onload = null; // Can be removed when TRIDUP-5616 is implemented.

      $nav.hover(makeNavWhite, makeNavTransparent);
      $(window).on('scroll', onScroll);

      equalizeHeights($cards.find('.tt-homepage__card__body'));
      makeNavTransparent();
      onScroll();

      function onScroll() {
         var scroll = window.scrollY;

         makeMenuSticky(scroll);
         makeCardsSticky(scroll)
      }

      function makeMenuSticky(scroll) {
         if (scroll > 80) {
            makeNavWhite();
         } else if (scroll <= 80) {
            makeNavTransparent();
         }
      }

      function makeCardsSticky(scroll) {
         var cardTitleHeight = 80;
         var cardDefaultTop = -20;
         var imgBottom = $cat.height();
         var cardDefaultPos = imgBottom + cardDefaultTop;
         var winHeight = window.innerHeight;
         var theFold = scroll + winHeight;
         var theCutoff = theFold - cardTitleHeight;

         if (cardDefaultPos > theCutoff) {
            $cards.css({top: theCutoff - imgBottom});
         } else {
            $cards.attr('style', '');
         }
      }

      function makeNavWhite() {
         $nav.removeClass('tt-main-navigation-module--transparent');
      }

      function makeNavTransparent() {
          var isPanelOpen = $panels.hasClass('in');
         var isScrolledDown = window.scrollY > 80;
         if (isPanelOpen || isScrolledDown) { return; }

         $nav.addClass('tt-main-navigation-module--transparent');
      }

      function equalizeHeights($elements) {
         var maxHeight = 0;
         $elements.each(function(i, el) { maxHeight = Math.max(el.clientHeight, maxHeight) } );

         $elements.height(maxHeight);
      }
   });

   // Mobile only
   onReady(function() {
      var isDesktop = window.innerWidth >= 1024;
      if (isDesktop) {
         return;
      }

      var $sports = $('.tt-homepage__cat#sports');

   onReady(function() {
      var time = 0.3;
      var ease = Quad.easeOut;

      var $sports = $('.tt-homepage__cat#sports');
      var $drive = $('.tt-homepage__cat#drive');
      var $sportsSubcats = $sports.find('.tt-homepage__subcats');
      var $driveSubcats = $drive.find('.tt-homepage__subcats');
      var $sportsSubcatItems = $sports.find('.tt-homepage__subcat');
      var $driveSubcatItems = $drive.find('.tt-homepage__subcat');
      var $sportsButton = $sports.find('.tt-homepage__button');
      var $driveButton = $drive.find('.tt-homepage__button');

      addEventListeners();

      function addEventListeners() {
         $sports.on('click', onClickSports);
         $drive.on('click', onClickDrive);
         $sportsSubcatItems.on('click', onClickSportsSubcatItem);
         $driveSubcatItems.on('click', onClickDriveSubcatItem);

         /* Detect if back-forward caching is used.
            This was evident in Safari, where only the clicked subcat would stay visible after hitting back. */
         $(window).on('pageshow', function(event) {
             if (event.originalEvent.persisted) {
               resetAllSubcatItems()
             }
         });
      }

      function onClickSports() {
         if($sportsSubcats.css('opacity') < 1) {
             var subcatOffset = '-50';
            TweenLite.to($sportsButton, time, { autoAlpha: 0, y: '-50', ease: ease });
            TweenLite.to($sportsSubcats, time, { autoAlpha: 1, y: subcatOffset, ease: ease });
         } else {
            TweenLite.to($sportsButton, time, { autoAlpha: 1, y: '0', ease: ease });
            TweenLite.to($sportsSubcats, time, { autoAlpha: 0, y: '0', ease: ease });
         }
         // Also close drive
         TweenLite.to($driveButton, time, { autoAlpha: 1, y: '0', ease: ease });
         TweenLite.to($driveSubcats, time, { autoAlpha: 0, y: '0', ease: ease });
      }

      function onClickDrive(e) {
         if($driveSubcats.css('opacity') < 1) {
            var subcatOffset = '-50';
            TweenLite.to($driveButton, time, { autoAlpha: 0, y: '-50', ease: ease });
            TweenLite.to($driveSubcats, time, { autoAlpha: 1, y: subcatOffset, ease: ease });
         } else {
            TweenLite.to($driveButton, time, { autoAlpha: 1, y: '0', ease: ease });
            TweenLite.to($driveSubcats, time, { autoAlpha: 0, y: '0', ease: ease });
         }
         // Also close sports
         TweenLite.to($sportsButton, time, { autoAlpha: 1, y: '0', ease: ease });
         TweenLite.to($sportsSubcats, time, { autoAlpha: 0, y: '0', ease: ease });
      }

      function onClickSportsSubcatItem(e) {
         $sportsSubcatItems.each(function(i, el) {
            if(el !== e.currentTarget) {
               TweenLite.to(this, 1, { autoAlpha: 0, ease: ease });
            }
         })

         e.stopPropagation();
      }

      function onClickDriveSubcatItem(e) {
         $driveSubcatItems.each(function(i, el) {
            if(el !== e.currentTarget) {
               TweenLite.to(this, 1, { autoAlpha: 0, ease: ease });
            }
         })

         e.stopPropagation();
      }

      function resetAllSubcatItems() {
         TweenLite.set($sportsButton, { autoAlpha: 1, y: 0 });
         TweenLite.set($driveButton, { autoAlpha: 1, y: 0 });
         TweenLite.set($sportsSubcats, { autoAlpha: 0, y: 0 });
         TweenLite.set($driveSubcats, { autoAlpha: 0, y: 0 });
         TweenLite.set($sportsSubcatItems, { autoAlpha: 1 });
         TweenLite.set($driveSubcatItems, { autoAlpha: 1 });
      }
   })
