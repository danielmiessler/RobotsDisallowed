	var BatasAnchor = $(".ants-newsticker-container").offset().top;
		
	if ($(this).scrollTop() >= BatasAnchor && $('#vbContainerMenu').css('position') != 'fixed') {
		$("#vbContainerMenu").find(".antsPopUpMenu").css({ "margin":"0 auto", "width":"1024px" });
		$("#vbContainerMenu").css({
			"position":"fixed", "top":"-80px", "opacity":"0", "border-bottom":"1px solid #ffffff" 
		}).animate({ "opacity":"0.98", "top":"0px" });
		
		$(".antsBackToTop").css({ "opacity":"0" }).show().animate({ "opacity":"0.5" });
	} 
	else if ($(this).scrollTop() < BatasAnchor && $('#vbContainerMenu').css('position') != 'relative') 
	{
		$("#vbContainerMenu").find(".antsPopUpMenu").css({ "margin":"0px", "width":"auto" });
		$('#vbContainerMenu').css({
			"position": "relative",
			"opacity": "1",
			"border-bottom":"none"
		});
		
		$(".antsBackToTop").hide();
	}
});
