function isSelfLoc(src){

	var result = false;
		
	/** parse out domain to make sure it points to our own **/
	if(src.substring(0, 7) == "http://" || src.substring(0, 8) == "https://") {
		var token = "://";
		var index = src.indexOf(token);
		var part = src.substring(index + token.length);
		var domain = "";
		
		/** parse up to end, first slash, or anchor **/
		for(var i=0; i<part.length; i++) {
			var character = part.charAt(i);
			
			if(character == '/' || character == ':' || character == '#') {
				break;
			} 
			else {
				domain += character;
			}
		}
		
		if(document.domain == domain){
			result=true;
		}		
	}
		
	return result;
}
	
function isXFSWhiteListed(parentLoc){   
	try {   
    	var result=false;
    	var XFSWhitelist="";
    	
    	if(XFSWhitelist.toLowerCase() == "null" || XFSWhitelist == "" || XFSWhitelist.length==0 ){
		    return false;
	    }
		if(parentLoc.toLowerCase() == "null" || parentLoc== "" || parentLoc.length==0 ){
		    return false;
	    }

	    if(isSelfLoc(parentLoc)){
		    return true;
	    }
	    
		var allowedLocations=XFSWhitelist.split("||");
		var allowedDomains = allowedLocations[0].split(" ");
		var isDomainAllowed = false;
		for (index = 0; index < allowedDomains.length; index++) {
			var pos = parentLoc.substring(8).indexOf("/");
			var parentDomain = parentLoc.substring(0, pos);
			if (parentDomain.equals(allowedDomains[index]) == 0) {
				isDomainAllowed = true;
				break;
			} 
		}
		
		if (isDomainAllowed){
			if(allowedLocations[1] == "*"){
				result=true;
			}
			else {  
				var allowedPages = allowedLocations[1].split("^");
				for (index = 0; index < allowedPages.length; ++index) {
					if(parentLoc.contains(allowedPages[index]) ) {
						result = true;
					}
				}
			}          
		}			
		return result;	   
    }   
    catch (err) {   
		return false;   
    }   
}
