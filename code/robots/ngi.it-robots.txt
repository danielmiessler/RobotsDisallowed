            $(function(){
            
            	// $("#chat-form-submit").prop('disabled', true);
            	
            	$(".chat-privacy-toggler").on("click", function(e){
            		e.preventDefault();
            		if ($(".chat-privacy-input").val() == "true") {
            			$(this).find("i").addClass("fa-square-o").removeClass("fa-check-square-o");
            			$(".chat-privacy-input").val("");

            			$(".chat-privacy-hint").show();
            			
            		} else {
	            		$(this).find("i").addClass("fa-check-square-o").removeClass("fa-square-o");

	            		// $("#chat-form-submit").prop('disabled', false);
	            		$(".chat-privacy-hint").hide();
            		}
            	});
            	
            	function setChatModalExecutionError(errorCode, errorMessage) {
            		$("#chatModal form").remove();
            		$("#chatModal .errorMessage").removeClass("hidden");
            	}
            	
            	function setChatModalValidationError(errorCode, errorMessage) {
            		$("#chatModal .warningMessage").removeClass("hidden");
            	}
            	
            	function setChatModalSuccess() {
            		$("#chatModal form").remove();
            		$("#chatModal .successMessage").removeClass("hidden");
            	}
            	
            	$("#chat-modal-form").on("submit", function(e){
            		e.preventDefault();
            		if ($(".chat-privacy-input").val() != "true") {
            			var shake = false;
            			if ($(".chat-privacy-hint").css("display") == "block") {
            				shake = true;
            			}
            			$(".chat-privacy-hint").show();
            			if (shake) {
            				$(".chat-privacy-hint").addClass("animated shake");
            				$(".chat-privacy-hint").one('webkitAnimationEnd mozAnimationEnd MSAnimationEnd oanimationend animationend', function(){
            					$(".chat-privacy-hint").removeClass("animated shake");
            				});
            			}
            			return;
            		} 
            		
            		var userName = $("#chat-user-name").val();
            		var userPhone = $("#chat-user-phone").val();
            		
            		$("#chat-user-name").prop('disabled', true);
            		$("#chat-user-phone").prop('disabled', true);
            		$("#chat-form-submit").prop('disabled', true);
            		$("#chat-form-submit").html('<span class="glyphicon glyphicon-refresh glyphicon-refresh-animate"></span> Loading...');
            		
            		$.ajax({
						url : '/home.html?ajaxMode=true&ajaxAction=callMeBack',
						type : 'get',
						dataType :	'json',
						async : true,
						cache : false,
						crossDomain : true,
						data : {
							"userName": userName, 
							"userPhone": userPhone
						},
						error : function (jqXHR, textStatus, error) {
							setChatModalExecutionError();
						},
						complete : function(){
							$("#chat-user-name").prop('disabled', false);
		            		$("#chat-user-phone").prop('disabled', false);
		            		$("#chat-form-submit").prop('disabled', false);
		            		$("#chat-form-submit").html("Invia");
						}, 
						success : function (data, textStatus, jqXHR) {
						
							var validationResult = data.validationResult;
							var executionResult = data.executionResult;
							
							if (executionResult == "ok" && validationResult == "ok") {
								setChatModalSuccess();
							} else {
								if (validationResult == "ok") {
									var errorCode = data.errorCode;
									var errorMessage = data.errorMessage;
									setChatModalExecutionError(errorCode, errorMessage);
								} else {
									var errorCode = data.errorCode;
									var errorMessage = data.errorMessage;
									setChatModalValidationError(errorCode, errorMessage);
								}
							}
								
						}
					});
            	});
            });
