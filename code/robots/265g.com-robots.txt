(function($){
	$.fn.hoverDelay = function(_opt){
		var defaults = {
			hoverDuring: 300,
			hoverTop: 0,
			hoverBottom: -170,
			hoverDom: 'li',
			hoverEvent: function(t, s){
				this.actEvent(t, s);
			},
			outEvent: function(t, s){
				this.actEvent(t, s);
			},
			actEvent: function(_this,size){
				_this.find('.ad_top_m').stop(false,true).animate({ top : size+'px'}, 300);
			}
		};
		var sets = $.extend(defaults,_opt || {});
		var hoverTimer, outTimer;
		$(this).find('.ad_top_m').css('top','-170px');
		return $(this).each(function(){
			$(this).find(sets.hoverDom).css('cursor','pointer');
			$(this).find(sets.hoverDom).hover(function(){
				_th = $(this);
				clearTimeout(outTimer);
				hoverTimer = setTimeout(function(){sets.hoverEvent(_th, sets.hoverTop);}, sets.hoverDuring);
			},function(){
				clearTimeout(hoverTimer);
				sets.outEvent($(this), sets.hoverBottom);
			});
		});
	}
})(jQuery);
