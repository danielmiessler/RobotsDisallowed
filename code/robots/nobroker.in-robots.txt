userType="phone";
}else if($('#userName').val().trim().match('(^[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,4}$)')!=undefined){
userType="email";
}
var params = $("#user_new").serialize();
params = params+"&userType="+userType;
performSignIn(params);
}
}
