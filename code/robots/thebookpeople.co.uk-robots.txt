    function ellipsise(element, titleWidth) {
      return element.each(function(){
          var text = $(this).html();
          var t = $(this.cloneNode(true)).hide().css({
                        'position': 'absolute',
                        'width': 'auto',
                        'overflow': 'visible',
                    });

          $(this).after(t);
          var proportion = titleWidth * 1.3 / t.width();
          t.remove();

          if(proportion < 1) {
            $(this).html(text.substr(0, text.length * proportion) + "...");
          }
      });
    }

        Handlebars.registerHelper('ifNotDefined', function(testValue, options) {
          // Return false if test value is defined and not null

          return options.fn(this);
        });

          // Return false if test value is non-numeric
          if (isNaN(parseFloat(testValue))) {
            return options.inverse(this);
          }
          return options.fn(this);
        });
        Handlebars.registerHelper('saveValue', function(isbn, price, rrp, rrp_eur) {
          // Return the difference between the item price and rrp, if positive
          if (rrp !== undefined && rrp !== null) {
            
            var retailPrice = rrp; 
            
          
            var currentIsbn = "";
            // Use the offer price if this is the current item
            var itemPrice = (isbn === currentIsbn) ? "" : parseFloat(price);
            var difference = parseFloat(retailPrice) - parseFloat(itemPrice);
            if (!isNaN(difference) && difference > 0) {
              return "Save &pound;" + difference.toFixed(2);
            }
          }
          return "";
        });
        Handlebars.registerHelper('addValue', function(items) {
          // Return the sum of the array, if positive
          var price = 0.00;
          var saving = 0.00;
          var currentIsbn = "";
          for ( var i=0, l=items.length; i < l; i++ ) {
            var p = (items[i].isbn === currentIsbn) ? "" : items[i].price;
            var r = false ? items[i].rrp_eur : items[i].rrp;
            if ( p !== undefined && p !== null ) {
              var unitPrice = parseFloat(p);
              if ( !isNaN(unitPrice) ) {
                price += unitPrice;
              }
            }
            if ( r !== undefined && r !== null ) {
              var unitRRP = parseFloat(r);
              if ( !isNaN(unitRRP) ) {
                saving += (unitRRP - unitPrice);
              }
            }
          }
          return "Total price:&nbsp;&nbsp;<span class=\"price\">&pound;"+price.toFixed(2)+"</span>"+
                 "<span class=\"saving-container\"><span class=\"save\">" + 
                 "Total saving:&nbsp;</span><span class=\"save save-price\"> &pound;"+saving.toFixed(2)+"</span></span>";      
        });
        Handlebars.registerHelper('addMultibuyValue', function(items) {
          var price = 0.00;
          var rrp = 0.00;
          for ( var i=0, l=items.length; i < l; i++ ) {
            var p = items[i].price;
            var r = items[i].rrp;
            if ( p !== undefined && p !== null ) {
              var unitPrice = parseFloat(p);
              if ( !isNaN(unitPrice) ) {
                price += unitPrice;
              }
            }
            if ( r !== undefined && r !== null ) {
              var unitRRP = parseFloat(r);
              if ( !isNaN(unitRRP) ) {
                rrp += unitRRP;
              }
            }
          }

          'currencyCode': 'GBP',
          'impressions': productArray
        }
      });
    }

    function getPageName(){
      var pageId = $('body').attr('id');
      var pageName = "";

      if (pageId == "page-browse-landingHome" || 
          pageId == "page-browse-landingSchools") {
        pageName = "Home";
      }
      else if (pageId == "page-browse-product") {
        pageName = "Product";
      }
      else if (pageId == "page-browse-searchListing") {
        pageName = "Search";
      }
      else if (pageId.startsWith("page-added-")) {
        pageName = "Added to Basket";
      }
      else if (pageId.startsWith("error")) {
        pageName = "Error";
      }
      else {  
        pageName = $('ul.breadcrumb > li').last().find('span').text();
        if (pageName == "") {
          pageName = $('ul.breadcrumb > li').last().text();
          if (pageName == "") {
            pageName = $('ul.breadcrumb > li').find('a > span').last().text();
          }
        }
      }
      return (pageName == "" ? document.title : pageName);
    }

function updateURLParameter(url, param, paramVal)
{
	var TheAnchor = null;
	var newAdditionalURL = "";
	var tempArray = url.split("?");
	var baseURL = tempArray[0];
	var additionalURL = tempArray[1];
	var temp = "";

	if (additionalURL)
	{
		var tmpAnchor = additionalURL.split("#");
		var TheParams = tmpAnchor[0];
		TheAnchor = tmpAnchor[1];
		if(TheAnchor)
			additionalURL = TheParams;

		tempArray = additionalURL.split("&");

		for (i=0; i<tempArray.length; i++)
		{
			if(tempArray[i].split('=')[0] != param)
			{
				newAdditionalURL += temp + tempArray[i];
				temp = "&";
			}
		}
	}
	else
	{
		var tmpAnchor = baseURL.split("#");
		var TheParams = tmpAnchor[0];
		TheAnchor  = tmpAnchor[1];

		if(TheParams)
			baseURL = TheParams;
	}

	if(TheAnchor)
		paramVal += "#" + TheAnchor;

	var rows_txt = temp + "" + param + "=" + paramVal;
	return baseURL + "?" + newAdditionalURL + rows_txt;
}
