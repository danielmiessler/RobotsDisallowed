	function changeElement(tipe){
		$("#pageLink1").attr("onClick","gantiPage('"+tipe+"','1','smallthumb');meAktive('pgGal',this);")
		$("#pageLink2").attr("onClick","gantiPage('"+tipe+"','2','smallthumb');meAktive('pgGal',this);")
		$("#pageLink3").attr("onClick","gantiPage('"+tipe+"','3','smallthumb');meAktive('pgGal',this);")
		$("#pageLink4").attr("onClick","gantiPage('"+tipe+"','4','smallthumb');meAktive('pgGal',this);")
	}
	
	function gantiPage(tipe,page,target){
		var div="#"+target;
		dataString="tipe="+tipe+"&page="+page;
		if($(".mainbanner").hasClass("buka")){
		}else{
			$(".mainbanner").animate({marginLeft:'935px'},500);
			$(".mainbanner").addClass("buka");
			$(".openme").css("display","");
		}
		$.ajax({
			 type: 'GET',
			 url: "http://www.fotografer.net/ajax/ajax.muka.php",
			 data: dataString,			 
			 success: function(data) {
				$(div).empty().append(data);
			}
		});				
	}
	
	function meAktive(trg,me){
		$("."+trg).removeClass("active");
		$(me).addClass("active");
	}
	
	function getMe(src,path){
		$("#thumbPreviewImg").attr("src",src);
		$("#thumbPreview").attr("href",path);
	}

	function slideme(){
		if($(".mainbanner").hasClass("buka")){
			$(".mainbanner").animate({marginLeft:'0px'},500);
			$(".mainbanner").removeClass("buka");			
			$(".openme").css("display","none");
		}else{
			$(".mainbanner").animate({marginLeft:'935px'},500);
			$(".mainbanner").addClass("buka");
			$(".openme").css("display","");
		}		
	}
	
	
	function setAktif(target){				
		$(".navmenu a").removeClass("active").addClass("");
		$(target).removeClass("").addClass("active");
	}
	
	function changePath(target){
		$("#FotoTerbaru_biggestFotoContainer a").attr("href",target);
	}
