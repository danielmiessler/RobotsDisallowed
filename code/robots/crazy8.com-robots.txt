function isValid(emailVal){
  var error = "0";
  	//Check for @
    if (emailVal.indexOf("@") == -1) {
		  error = "1";
	} else if (emailVal.indexOf(".") == -1) {//Check for .
		  error = "1";
	} else {
		  var size = emailVal.length;
		  for(var i = 0; i < size; i++){
			  ch = emailVal.substring(i, i + 1);
			  if ((ch >= "A" && ch <= "Z") || (ch >= "a" && ch <= "z") || (ch == 	"@") || (ch == ".") || (ch == "_") || (ch == "-") || (ch >= 		"0" && ch <= "9")) {
				 continue;
			  }else{
				 error = "1";
				 break;
			  }
		}
	}
  
	if (error == "1")
	  return false;

	return true;
}

function checkDate(inDay, inMon, inYr)
{
	var toDate = new Date();
	var isDayEntered = false;
	var checkDate;
	
	if(inDay == 0)
	{
		checkDate = new Date(inYr, inMon, toDate.getDate());
	}
	else
	{
		checkDate = new Date(inYr, inMon, inDay);
		isDayEntered = true;
	}
	
	if(isDayEntered)
	{
		var diff = (toDate - checkDate);
		if(diff>=0)
		{
			alert("please enter valid date" + checkDate);
			return false;
		}
	}
	else
	{
		var chYear = checkDate.getUTCFullYear();
		var toYear = toDate.getUTCFullYear();
		var sameYear = false;
		
		if(chYear-toYear < 0)
		{
			alert("please enter valid date" + checkDate);
			return false;
		}
		else if(chYear-toYear == 0)
			sameYear = true;
			
		var chMon = checkDate.getUTCMonth();
		var toMon = toDate.getUTCMonth();
	
		if(sameYear && (chMon-toMon<0))
		{
			alert("please enter valid date " + checkDate);
			return false;
		}
	}

	return true;
}
