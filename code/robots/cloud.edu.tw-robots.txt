// 使用於強制讀取新頁面，避免cache
function urlWithTimestamp(url) {
	if(true) return url;
	var qIndex = url.indexOf("?");
	if (qIndex!=-1) {
		if (url.substring(qIndex+1).length>0) {
			url += "&";
		}
		url += "t="+new Date().getTime();
	} else {
		url += "?t="+new Date().getTime();
	}
	return url;
}
function loadContent(url, doPushState) {
	var useNewTech = false;
	if(useNewTech) {
		if(typeof(doPushState)=="undefined") {
			doPushState = true;
		}
		$.get(url, function(html) {
			if(doPushState) {
				url = urlWithTimestamp(url);
				history.pushState({url:url}, "", url);
			}
			$("#mainContent").html(html);
			$(document).scrollTop(0);
		});
	} else {
		location.href = url;
	}
}

function changeIdentity(im) {
	var u = window.location.search;
	var s = '';
	if(u.indexOf("?") != -1) {
		var hasIm = false;
		var getSearch = u.split("?");
	    var getPara = getSearch[1].split("&");
	    for (var i=0;i<getPara.length;i++) {
	    	var para = getPara[i];
	      	var paraVal = para.split("=");
	      	if(i==0) {
	      		s = '?';
	      	} else {
	      		s += '&';
	      	}
	      	s += paraVal[0] + '=';
	      	if(paraVal[0] == 'im') {
	      		s += im;
	      		hasIm = true;
	      	} else {
	      		s += paraVal[1];
	      	}
	    }
	    if(!hasIm) {
			s += '&' + 'im=' + im;
	    }
	} else {
		s = '?im=' + im;
	}
	window.location.search = s;
}

function loginSuccess(data){

		if (data.url!='') {
			var uid = data.uid;
			uid = typeof(uid)=="undefined"?"":uid;
			//沒有UID為註冊 有為登入
			if(uid!=''){
				$("#ssoLoginForm").attr("action", data.url);
				$("#loginUid").val(uid);
				$("#loginPassword").val(data.password);
				$("#loginAccountType").val(data.accountType);

				$("#ssoRegisterForm").attr("action", data.url);
