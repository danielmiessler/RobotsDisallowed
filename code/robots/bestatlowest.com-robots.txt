    function IsAlpha(obj) {
        var digits = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ. ";
        var temp;
        var resultt = false;
        if (obj != "") {
            for (var i = 0; i < obj.length; i++) {
                temp = obj.substring(i, i + 1);
                if (digits.indexOf(temp) == -1) {
                    resultt = false;
                    break;
                }
                else {
                    resultt = true;
                }
            }
        }
        return resultt;
    }
