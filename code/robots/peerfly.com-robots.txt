    $(document).ready(function() {
        $('.PFRotate').each(function(i, e){
            var speed = $(e).data('speed');
            setInterval(function() { PFRotate($(e)); }, speed);
        });
    });
    function PFRotate(element){
        $element = $(element);
        var nextPosition = $element.data('active')+1;
        $element.children('div').hide();

        if (nextPosition > $element.children('div').length)
        {
            var nextPosition = 1;
        }

        $element.children('div:nth-child('+nextPosition+')').fadeIn();
        $element.data('active', nextPosition);
    }
</script>
