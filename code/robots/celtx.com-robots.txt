	function checkFormSectionValidity($form, $section){
		var valid = true;
		$req = $section.find('input[required], select[required], textarea[required]');
		$req.each(function(){
			if(!this.validity.valid){
				valid = false;
				$(this).addClass('input-error');
				$(this).one('change', function(){
					$(this).removeClass('input-error');
				});
			}
		});

		if(!valid){
			//trigger validation popups

			$form.one('submit', function(e){
				e.preventDefault();
			});

			$inputSubmit = $form.find('input[type="submit"]');

			if($inputSubmit.length>0){
				$inputSubmit.click();
			} else {
				$form.submit();
			}
		}

		return valid;
	}

// Fixed elements do not work on mobile devices, thus must change some behavior when user is on a mobile device
function _isMobile(){
    var isMobile = (/iphone|ipod|android|ie|blackberry|fennec/).test
         (navigator.userAgent.toLowerCase());
    return isMobile;
}

// Returns the position (either top or left) of an element
function getOffset(el) {
    var _x = 0;
    var _y = 0;
    while( el && !isNaN( el.offsetLeft ) && !isNaN( el.offsetTop ) ) {
        _x += el.offsetLeft - el.scrollLeft;
        _y += el.offsetTop - el.scrollTop;
        el = el.offsetParent;
    }
    return {top: _y, left: _x};
}

// Place the signup button in the navbar directly above the signup button in the dropdown or else according to window size
function placeSignUpBtn() {
	$navButton = document.getElementById('narrowSignup');
		var x;
		if ($(window).width() > 804) { 
			x = getOffset( document.getElementById('fakeSignUp') ).left;
			$navButton.style.left = x + "px";
		}
		else {
			x = $(window).width();
			x = x - 172;
			$navButton.style.left = x + "px";
		}
}

var fadeDistance = 125;

function _isMobile(){
    var isMobile = (/iphone|ipod|android|ie|blackberry|fennec/).test
         (navigator.userAgent.toLowerCase());
    return isMobile;
}

// Fade out header text and signup fields on scroll
function fadeHeaderText() {

  	var array = ["heading", "subhead1", "subhead2", "signUp1", "signUp2", "signUp3", "signUp4", "signUp5"];


  	for (var i = 0; i < array.length; i++) {

  		var string = "#" + array[i];
  		$(string).css("visibility", "hidden");
  		
  		if (string == "#signUp5") {
  			if ($(window).width() >= 805) {
  				var x = getOffset( document.getElementById(array[6]) ).top;
  			}
  			else {
  				var x = getOffset( document.getElementById("signMeUp")).top;
  			}
  		}
  		else {
  			var x = getOffset( document.getElementById(array[i]) ).top;
  		}

	  	if (x < fadeDistance) {
	  		$(string).addClass('scrollFadeOut');
	  		$(string).removeClass('scrollFadeIn');
	  		$(string).css("visibility", "hidden");
	  	}
	  	else {
	  		$(string).css("visibility", "visible");
	  		$(string).addClass('scrollFadeIn');
	  		$(string).removeClass('scrollFadeOut');
	  	}

  	}

  	// If window size is small, have sign up button fade instead of slide
  	if ($(window).width() < 805) {
  		$("#signMeUp").css("visibility", "hidden");
  		var x = getOffset( document.getElementById("signMeUp")).top;
  		if (x < fadeDistance) {
	  		$("#signMeUp").addClass('scrollFadeOut');
	  		$("#signMeUp").removeClass('scrollFadeIn');
	  		$("#signMeUp").css("visibility", "hidden");
	  	}
	  	else {
	  		$("#signMeUp").css("visibility", "visible");
	  		$("#signMeUp").addClass('scrollFadeIn');
	  		$("#signMeUp").removeClass('scrollFadeOut');
	  	}
  	}
  	else {
  		$("#signMeUp").css("visibility", "visible");
	  	$("#signMeUp").addClass('scrollFadeIn');
	  	$("#signMeUp").removeClass('scrollFadeOut');
  	}


}

function setUpAnchorScrolls(){
	jQuery(document).ready(function($) {

		$anchors = $('a[href^=#]');
		$anchors.click(function(e){
			var jump = $(this).attr('href').substring(1);

			if(jump.length <= 0){
	        	return;
	        }

			e.preventDefault();

			$el = $('*[name='+jump+']');
		    $el.addClass('focusbg');
		     setTimeout(function(){
		    	$el.addClass('focusbg-start');
			}, 400);
	        var new_position = $el.offset();
	         $('html, body').animate({
		         scrollTop: (new_position.top-50)
		     }, 1500, function(){
		     	$el.find('input:first').focus();
		     	setTimeout(function(){
		     		$el.removeClass('focusbg-start');
		     		setTimeout(function(){
		     			$el.removeClass('focusbg');
		     		}, 1000);
		     	}, 1000);
		     });

		});
	});
}

$(document).ready(function(){

	$container = document.getElementById('topBarContainer');
	$container.style.position = "absolute";

	// Signup button in the header is intitially hidden on the landing page
	$("#narrowSignup").hide();
	// Nav bar is intially transparent on the landing page
	$('#topBar').removeClass('topBar-fixed');

	setUpAnchorScrolls();

	// Header text does not fade if on mobile device
	var mobile = _isMobile();
    if (!mobile) {
    	fadeHeaderText();
    }

	window.onscroll = function(){

		// Switch to thinner signup button on scroll
		var z = getOffset( document.getElementById('signMeUp') ).top;
		$headerButton = document.getElementById('buttonFade');
		if (z < 10) {
			$("#narrowSignup").show();
			$headerButton.style.opacity = "0";
			
		} else {
			$("#narrowSignup").hide();	
			$headerButton.style.opacity = "1";
		}

		// When scrolled past bottom of header, change background of nav bar from transparent to an image
		// Also change the functionality of the sign up button
		var scrollPosition = $(window).scrollTop();
	    if (scrollPosition >= ($('.crewfull').height() - $('#topBar').height())) {
	    	placeNavImage();
	    	$('#topBar').addClass('topBar-fixed');
	    	document.getElementById("narrowSignup").onclick = function () { dropHeader(); };
	    }
	    else{
			$('#topBar').removeClass('topBar-fixed');
			document.getElementById("narrowSignup").onclick = function () { scrollUp(); };
	    }

	    if (!mobile) {
	
			fadeHeaderText();

		 	// Change the functionality of the sign up button for after the fields disappear but before the dropdown will function correctly
		 	// Button will now cause a scroll to top of screen
		 	var distanceFromTop = getOffset( document.getElementById("signUpForm") ).top;
			if (distanceFromTop < fadeDistance) {
				document.getElementById("changeFunction").style.display = "block";                   
			}
			else {
				document.getElementById("changeFunction").style.display = "none";
			}
	    }

	};

});
