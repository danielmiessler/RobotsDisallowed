			requstURL = context+"/includes/checkLogin.jsp"
		}
		//SCLVAT-1578 Optimize customer login check
    	$.ajax({
        	url : requstURL,
        	async : false,
        	success : function (data){
        		loginStatus = data;
        	}
    	});
		var isCustomerLogin = eval(loginStatus.split("|")[0]);
		if(!isCustomerLogin){
			top.window.location.href=context+"/myaccount/includes/loginForm.jsp"
			return false;
		}

	}
