<script>

function mobNavToggle(){
  var slideAmt = $( "#mobNav" ).width();

  // mobile nav is showing, so hide it
  if ($( "#mobNav-blocker" ).is(":visible")) {

    // set all of the page elements positions back to relative
    $( "#wrapperPrimary" ).css('position','relative');

    // put all of the page elements back to their default location
    $( "#mobNav" ).animate({left:-slideAmt + "px"}, 300);
    $( "#wrapperPrimary" ).animate({left:"0px"}, 300);
    $( "#mobNavToggleBtn" ).animate({left: "10px"}, 300);


    // match the width and height of the blocker to the width and height of the page
    $( "#mobNav-blocker" ).css("height", $( document ).height() + "px");
    $( "#mobNav-blocker" ).css("width", $( document ).width() + "px");

    // hide the blocker and the nav
    $( "#mobNav-blocker" ).animate({opacity:0}, {duration:300, complete:function() {
      $( "#mobNav-blocker" ).css('display','none');
      $( "#mobNav" ).css('display','none');
    }});

  // mobile nav is hidden, so show it
  } else {        
    // match the width and height of the blocker to the width and height of the page
    $( "#mobNav-blocker" ).css("height", $( document ).height() + "px");
    $( "#mobNav-blocker" ).css("width", $( document ).width() + "px");

    // show the blocker and the nav
    $( "#mobNav-blocker" ).css('display','block');
    $( "#mobNav-blocker" ).animate({opacity:0.8}, 300);

    // set all of the page elements positions to fixed so that only the nav will scroll
    $( "#wrapperPrimary" ).css('position','fixed');

    // slide all of the page elements to allow room for the nav
    console.log("$('html').offset().top: " + $(document).scrollTop())
    $( "#mobNav" ).css('top', $(document).scrollTop());
    $( "#mobNav" ).css('display','block');
    $( "#mobNav" ).animate({left: "0px"}, 300);
    $( "#wrapperPrimary" ).animate({left: slideAmt + "px"}, 300);
    $( "#mobNavToggleBtn" ).animate({left: (slideAmt + 10) + "px"}, 300);
  }

  $( window ).resize(function() {
    if ($( "#mobNav-blocker" ).is(":visible")/* && $( "#mobNav-blocker" ).css("opacity") == 0.8*/) {
      $( "#mobNav-blocker" ).css("height", $( document ).height() + "px");
      $( "#mobNav-blocker" ).css("width", $( document ).width() + "px");
    }
  });
}

$(document).ready(function () { 
  var ulNum = 0;

  $("#mobNav .menu.nav .expanded.dropdown ul").each(function() {
    console.log("top nav - " + $(this).attr("class"))
      $( "<a id=\"mobNav-dropdown-toggle-" + ulNum + "\" class=\"mobNavClass\" data-toggle=\"dropdown\" data-target=\"#\" style=\"top:0px; right:-25px; position: absolute; cursor: pointer;\">+</a>" ).insertAfter( $(this) );
      $(this).css('display', 'none');
      $(this).addClass('mobNav-dropdown-menu-' + ulNum);
      ulNum++;
  });

  $("#mobNav .menu.nav .expanded.dropdown .dropdown-submenu ul").each(function() {
      $( "<a id=\"mobNav-dropdown-toggle-" + ulNum + "\" data-toggle=\"dropdown\" data-target=\"#\" style=\"top:0px; right:-25px; position: absolute; cursor: pointer;\">+</a>" ).insertAfter( $(this) );
      $(this).css('display', 'none');
      $(this).addClass('mobNav-dropdown-menu-' + ulNum);
      ulNum++;
  });

  var toggle;
  for (var i = 0; i <= ulNum; i++)
  {
    toggle = $("#mobNav-dropdown-toggle-" + i);
    (function (_toggle, _i) {
      _toggle.click(function () {
        console.log(".mobNav-dropdown-menu-" + _i)
        if ($('.mobNav-dropdown-menu-' + _i).is(":visible"))
        {
          $('.mobNav-dropdown-menu-' + _i).css('display', 'none');
        } else {
          $('.mobNav-dropdown-menu-' + _i).css('display', 'block');
        }
      });
    })(toggle, i);    
  }
  
});
