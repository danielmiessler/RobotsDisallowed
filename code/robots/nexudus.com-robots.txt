  function updateAccountCounter(){
    var count = Math.floor((new Date().getTime() - 1401455628718) / 1000 / 60 / 60 / 24);
    $({countNum: 1}).animate({countNum: 537 + count}, {
      duration: 3600,
      easing:'linear',
      step: function() {
        $('#accounts-count').text(Math.floor(this.countNum));
      },
      complete: function() {
        $('#accounts-count').text(this.countNum);
      }
    });
  }
