        function AutoComplete_OnClientShowing(behaviour, e) {
            var ResultsDiv = behaviour.get_completionList();
            var oldCategory = "";
            var html = "";

            //for each result
            for (var i = 0; i < ResultsDiv.childNodes.length; i++) {
                var item = ResultsDiv.childNodes[i];
                //since the pair object stores the values to display in the second variable, get it out to do our own processing
                if (item != null && item._value != null && item._value.Second != null){
                    var vals = item._value.Second;
                    item.setAttribute("class", "result_div");
                    //remove all child nodes
                    while (item.childNodes.length > 0)
                        item.removeChild(item.firstChild);
                    //build html for result links
                    //if this is the view all link, markup is different
                    if (vals[5] == "acr_viewall")
                        html = "<a class='acr_viewall' href='#' title='" + vals[1] + "'><b>" + vals[0] + "</b></a>";

                        html = "<ul class='" + vals[6] + "'><li><a class='acr_link' href='" + vals[4] + "' title='" + vals[1] + "'><b>" + vals[0] + "</b></a><p title='" + vals[3] + "'>" + vals[2] + "</p></li></ul>";
                    //if new category and not view all link, add category before link
                    if (oldCategory != vals[5] && vals[5] != "acr_viewall") {
                        //get this category to compare for next time
                        oldCategory = vals[5];
                        html = "<div class='acr_category'>" + vals[5] + "</div>" + html;

                }
