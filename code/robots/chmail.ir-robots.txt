    $(document).ready(function($) {

            setTimeout(function () {
                location.replace("/cas")
            }, 4000)
        }
        if(error) {
            Materialize.toast(error, 8000)
            $('.toast').addClass('red').parent().css({bottom:0,left:'8%',top:'inherit'})
            
        }
        
        setTimeout(function (argument) {
          // user name value
          var usernameValue = document.getElementById('username').value

          if(usernameValue) return
          $('.tap-target').delay('3000').tapTarget('open')
          setTimeout(function (argument) {
            $('.tap-target').delay('3000').tapTarget('close')
          }, 2000) 
        }, 4500)    
        
        
        var changeInputDirByTyping = function (el) {
            var x =  new RegExp("[\x00-\x80]+") // is ascii
            var element = $(el)
            var isAscii = x.test(el.value[0])
            if(isAscii)
            {
                element.css("direction", "ltr")
                element.css("textAlign", "left")
                element.css("fontFamily", "Arial")
            }
            else
            {
                element.css("direction", "rtl")
                element.css("textAlign", "right")
                element.css("fontFamily", "iranSans")
            }
            
        }
        $('.input-field > input').on('input keypress', function(event){
            if(event.currentTarget.value.length === 1)
            {
                changeInputDirByTyping(event.currentTarget)
            }
        })
        $('.input-field > input').on('paste', function(event){
           setTimeout(function () {
                changeInputDirByTyping(event.currentTarget)
          }, 100);
        })
        $('.input-field > input').each(function(index, el) {
            if (el.value === '') {
                $(el).css({
                    direction: 'rtl',
                    textAlign: 'right',
                    fontFamily: 'iranSans'
                });
                return
            }
            changeInputDirByTyping(el)
        })
        if(window.innerWidth <= 1200) {
          $('.tap-target-wraper-wraper').css('left', '0');
        } else {
          $('.tap-target-wraper-wraper').parent().addClass('mobile-display-none')
        }

    })
</script>
